import axios from "axios";

axios.post(
  "/api/report/subquery/enterprise-risk",
  {
    columns: [
      "ENTERPRISE_NAME",
      "INDUSTRY_CODE",
      "TOTAL_REVENUE",
      "AVG_ACTIVE_USERS",
      "TOTAL_API_CALLS"
    ],
    enterpriseIds: [1, 2],
    orderBy: {
      key: "TOTAL_REVENUE",
      dir: "DESC"
    }
  },
  {
    headers: {
      "X-ROLE": "USER"
    }
  }
)
.then(res => {
  console.log("SUBQUERY RESULT:", res.data);
})
.catch(err => {
  console.error("SUBQUERY ERROR:", err.response?.data || err);
});
